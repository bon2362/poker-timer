# Settings Screen Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Добавить экран настроек в `poker-timer.html` — редактирование блайндов и временных параметров с сохранением в localStorage.

**Architecture:** Всё в одном файле `poker-timer.html`. `config` объект загружается из localStorage при старте (или берёт дефолты). `buildStages()` использует `config` вместо захардкоженных констант. Экран настроек — скрытый `<div id="settings-screen">`, показывается через CSS-класс `body.is-settings`.

**Tech Stack:** Vanilla HTML5, CSS3, JavaScript (ES6), localStorage

---

## File Structure

```
poker-timer.html   ← единственный файл, все изменения здесь
```

Текущие секции `<script>` (порядок после изменений):
1. `// ─── CONFIG ───` ← **новая секция** перед DATA
2. `// ─── DATA ───` ← `buildStages()` переписывается под `config`, `BLIND_LEVELS`/`LEVEL_DURATION`/etc убираются
3. `// ─── STATE ───` ← без изменений
4. `// ─── TIMER ENGINE ───` ← добавляется `rebuildStages()`
5. `// ─── AUDIO ───` ← без изменений
6. `// ─── HELPERS ───` ← `getNextInfo()` использует `config.breakDuration`
7. `// ─── RENDER ───` ← без изменений
8. `// ─── SETTINGS ───` ← **новая секция** в конце

```js
  ],

function loadConfig() {
  try {
    const saved = localStorage.getItem('pokerTimerConfig');
    return saved ? JSON.parse(saved) : structuredClone(DEFAULT_CONFIG);
  } catch {
    return structuredClone(DEFAULT_CONFIG);
  }
}

function saveConfig(cfg) {
  localStorage.setItem('pokerTimerConfig', JSON.stringify(cfg));
}

let config = loadConfig();

// ─── DATA ────────────────────────────────────────────────────────────────────

function buildStages() {
  const levelDuration = config.levelDuration * 60;
  const breakDuration = config.breakDuration * 60;
  const result = [];
  for (let i = 0; i < config.blindLevels.length; i++) {
    const levelNum = i + 1;
    result.push({
      type: 'level',
      levelNum,
      sb: config.blindLevels[i].sb,
      bb: config.blindLevels[i].bb,
      duration: levelDuration,
    });
    const isLast = levelNum === config.blindLevels.length;
    if (levelNum % config.breakEvery === 0 && !isLast) {
      result.push({ type: 'break', duration: breakDuration });
    }
  }
  return result;
}

let stages = buildStages();
```

```js
    // ─── SETTINGS ────────────────────────────────────────────────────────────────

    let formBlinds = []; // рабочая копия блайндов для формы

    function openSettings() {
      if (!isPaused) { isPaused = true; render(); }
      document.getElementById('cfg-level-duration').value = config.levelDuration;
      document.getElementById('cfg-break-duration').value = config.breakDuration;
      document.getElementById('cfg-break-every').value    = config.breakEvery;
      formBlinds = config.blindLevels.map(l => ({ sb: l.sb, bb: l.bb }));
      clearSettingsErrors();
      renderBlindsTable();
      document.body.classList.add('is-settings');
    }

    function closeSettings() {
      document.body.classList.remove('is-settings');
    }

    function getFormBreakEvery() {
      return Math.max(1, parseInt(document.getElementById('cfg-break-every').value, 10) || 1);
    }

    function renderBlindsTable() {
      const tbody   = document.getElementById('blinds-tbody');
      const bEvery  = getFormBreakEvery();
      tbody.innerHTML = '';
      formBlinds.forEach((level, i) => {
        const levelNum = i + 1;
        const tr = document.createElement('tr');
        tr.innerHTML = `
          <td class="row-num">${levelNum}</td>
          <td><input type="number" min="1" value="${level.sb}"
                     oninput="formBlinds[${i}].sb = +this.value"></td>
          <td><input type="number" min="1" value="${level.bb}"
                     oninput="formBlinds[${i}].bb = +this.value"></td>
          <td><button class="del-btn" onclick="removeBlindRow(${i})">✕</button></td>
        `;
        tbody.appendChild(tr);
        const isLast = i === formBlinds.length - 1;
        if (levelNum % bEvery === 0 && !isLast) {
          const divider = document.createElement('tr');
          divider.className = 'break-divider';
          divider.innerHTML = '<td colspan="4">── ☕ Перерыв ──</td>';
          tbody.appendChild(divider);
        }
      });
    }

    function addBlindRow() {
      const last = formBlinds[formBlinds.length - 1];
      formBlinds.push({ sb: last ? last.sb * 2 : 10, bb: last ? last.bb * 2 : 20 });
      renderBlindsTable();
    }

    function removeBlindRow(i) {
      formBlinds.splice(i, 1);
      renderBlindsTable();
    }

    function resetToDefaults() {
      document.getElementById('cfg-level-duration').value = DEFAULT_CONFIG.levelDuration;
      document.getElementById('cfg-break-duration').value = DEFAULT_CONFIG.breakDuration;
      document.getElementById('cfg-break-every').value    = DEFAULT_CONFIG.breakEvery;
      formBlinds = DEFAULT_CONFIG.blindLevels.map(l => ({ sb: l.sb, bb: l.bb }));
      clearSettingsErrors();
      renderBlindsTable();
    }

    function clearSettingsErrors() {
      document.getElementById('cfg-level-duration').classList.remove('error');
      document.getElementById('cfg-break-duration').classList.remove('error');
      document.getElementById('cfg-break-every').classList.remove('error');
      document.getElementById('err-level-duration').style.display = 'none';
      document.getElementById('err-break-duration').style.display = 'none';
      document.getElementById('err-break-every').style.display    = 'none';
      document.getElementById('err-blinds').style.display         = 'none';
      document.querySelectorAll('#blinds-tbody input').forEach(i => i.classList.remove('error'));
    }

    // Перерисовывать разделители при изменении поля "каждые N"
    document.getElementById('cfg-break-every').addEventListener('input', renderBlindsTable);
```

```js
    function saveSettings() {
      clearSettingsErrors();
      let valid = true;

      const levelDuration = parseInt(document.getElementById('cfg-level-duration').value, 10);
      const breakDuration = parseInt(document.getElementById('cfg-break-duration').value, 10);
      const breakEvery    = parseInt(document.getElementById('cfg-break-every').value, 10);

      if (!levelDuration || levelDuration < 1 || levelDuration > 999) {
        document.getElementById('cfg-level-duration').classList.add('error');
        document.getElementById('err-level-duration').style.display = 'block';
        valid = false;
      }
      if (!breakDuration || breakDuration < 1 || breakDuration > 999) {
        document.getElementById('cfg-break-duration').classList.add('error');
        document.getElementById('err-break-duration').style.display = 'block';
        valid = false;
      }
      if (!breakEvery || breakEvery < 1) {
        document.getElementById('cfg-break-every').classList.add('error');
        document.getElementById('err-break-every').style.display = 'block';
        valid = false;
      }

      if (formBlinds.length === 0) {
        document.getElementById('err-blinds').style.display = 'block';
        valid = false;
      } else {
        formBlinds.forEach((level, i) => {
          const inputs = document.querySelectorAll(`#blinds-tbody tr:not(.break-divider)`);
          if (!level.sb || level.sb <= 0 || !level.bb || level.bb <= 0) {
            if (inputs[i]) {
              inputs[i].querySelectorAll('input').forEach(inp => inp.classList.add('error'));
            }
            valid = false;
          }
        });
      }

      if (!valid) return;

      // Проверить активность игры
      const gameActive = currentStage > 0 || timeLeft < stages[0].duration;
      if (gameActive) {
        if (!confirm('Изменение настроек сбросит текущую игру. Продолжить?')) return;
      }

      // Применить настройки
      config = {
        levelDuration,
        breakDuration,
        breakEvery,
        blindLevels: formBlinds.map(l => ({ sb: l.sb, bb: l.bb })),
      };
      saveConfig(config);
      rebuildStages();
      restartTournament();
      closeSettings();
    }
```
